# 击鼓传花





```c++
#include <cstdio>
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <sstream>
using namespace std;

#define modval 1000000007

vector<vector<long long>> mul(vector<vector<long long>> a, vector<vector<long long>> b);
vector<vector<long long>> mymod(vector<vector<long long>> a, long long n);

int main()
{
	long long N, K;
	cin >> N >> K;
	if (N == 1)
	{
		cout << 0 << endl;
		return 0;
	}
	vector<vector<long long>> val(2);
	val.resize(2);
	val[0].resize(2);
	val[1].resize(2);


	val[0][0] = K - 1;
	val[0][1] = 0;
	val[1][0] = K - 1;
	val[1][1] = -1;

	val = mymod(val, N - 1);
	cout << val[1][0];
	return 0;

}


vector<vector<long long>> mul(vector<vector<long long>> a, vector<vector<long long>> b)
{
	vector<vector<long long>> c;
	c.resize(2);
	c[0].resize(2);
	c[1].resize(2);
	for (int i = 0; i < 2; i++)
	{
		for (int j = 0; j < 2; j++)
		{
			if (a[i][j] == 0) continue;
			for (int k = 0; k < 2; k++)
			{
				if (b[j][k] == 0) continue;
				c[i][k] += a[i][j] * b[j][k] % modval;
				if (c[i][k] > modval)
				{
					c[i][k] -= modval;
				}
				else if (c[i][k] < 0)
				{
					c[i][k] += modval;
				}

			}
		}
	}
	return c;
}

vector<vector<long long>> mymod(vector<vector<long long>> a, long long n)
{
	vector<vector<long long>> res;
	res.resize(2);
	res[0].resize(2);
	res[1].resize(2);

	res[0][0] = 1;
	res[1][1] = 1;
	while (n)
	{
		if (n & 1)
		{
			res = mul(a, res);
		}
		a = mul(a, a);
		n >>= 1;
	}
	return res;
}
```
