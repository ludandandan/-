# 单词接龙

```c++
#include <cstdio>
#include <vector>
#include <algorithm>
#include <cmath>
#include <sstream>
#include <unordered_map>
using namespace std;

bool solve(int num, char** arr);
int main()
{
	int num = 3;
	char a1[] = "dab";
	char a2[] = "dat";
	char a3[] = "dad";
	char *arr[3]= {a1, a2, a3};
	bool res = solve(num, arr);
	return 0;
}

bool solve(int num, char** arr)
{
	//用一个map记录每个单词的首尾字母出现的次数，
	//若是在头部出现就-1，若是在尾巴出现就+1，若是最后只剩下一个-1和一个+1就欧克
	unordered_map<char, int> map;
	for (int i = 0; i < num; i++)
	{
		string str = *arr + i;
		char head = str[0];
		char tail = str[str.size() - 1];
		if (map.count(head) == 0) map.insert({ head, -1 });
		else map[head]--;
		if (map.count(tail) == 0) map.insert({ tail, 1 });
		else map[tail]++;
	}

	//检查map里的正负1个数
	int zheng = 0;
	int fu = 0;
	for (auto w : map)
	{
		if (w.second == 0) continue;
		else if (w.second == 1) zheng++;
		else if (w.second == -1) fu++;
		else return false;
	}
	if ((zheng == 1 && fu == 1) || (zheng == -1 && fu == -1)) return true;
	else return false;
}
```
